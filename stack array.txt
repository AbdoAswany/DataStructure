#include <iostream>

using namespace std;
#define MAX_SIZE 100

class ArrayStack {
private:
    int stack[MAX_SIZE];
    int top;

public:
    ArrayStack() {
        top = -1;
    }

    bool isEmpty() {
        return top == -1;
    }

    bool isFull() {
        return top == MAX_SIZE - 1;
    }

    void push(int item) {
        if (isFull()) {
            std::cout << "Stack is full. Cannot push item " << item << ".\n";
            return;
        }

        top++;
        stack[top] = item;
        std::cout << "Item " << item << " pushed into the stack.\n";
    }

    int pop() {
        if (isEmpty()) {
            std::cout << "Stack is empty. Cannot pop from an empty stack.\n";
            return -1;
        }

        int item = stack[top];
        top--;
        std::cout << "Item " << item << " popped from the stack.\n";
        return item;
    }

    void displayStack() {
        if (isEmpty()) {
            std::cout << "Stack is empty.\n";
            return;
        }

        std::cout << "Stack: ";
        for (int i = top; i >= 0; i--) {
            std::cout << stack[i] << " ";
        }
        std::cout << std::endl;
    }
};

int main() {
    ArrayStack stack;

    stack.push(10);
    stack.push(20);
    stack.push(30);
    stack.displayStack();

    stack.pop();
    stack.displayStack();

    stack.pop();
    stack.pop();

    return 0;
}
