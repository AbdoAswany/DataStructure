#include <iostream>

using namespace std;

void displayArray(int arr[], int size) {
    for (int i = 0; i < size; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;
}

int main() {
    int nums[] = {1, 2, 4, 5, 6, 8};
    int size = sizeof(nums) / sizeof(nums[0]);

    displayArray(nums, size);

    return 0;
}
